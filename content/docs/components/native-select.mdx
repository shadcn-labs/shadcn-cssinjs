## Usage

```tsx
import {
  NativeSelect,
  NativeSelectOption,
} from "@/components/ui/native-select/native-select";

<NativeSelect>
  <NativeSelectOption value="apple">Apple</NativeSelectOption>
  <NativeSelectOption value="banana">Banana</NativeSelectOption>
</NativeSelect>
```
